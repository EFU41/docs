### Trigger AI Flow

<CodeGroup>

```js javascript
// Please make sure that you run 'npm install @eachlabs/aiflow'
import Each from '@eachlabs/aiflow';

const each = new Each({
    auth: process.env.EACH_API_KEY ||
    'YOUR_API_KEY'
});

const triggerID = await each.flow.trigger("flow-id", {
    parameters: {
        text: "Hello World" // parameters for the flow
    }
});

console.log(triggerID)
```

```python python
# We will launch our python client SDK soon.
import requests

# Make sure that you set EACH_API_KEY environment variable

response = requests.post(
    "https://flows.eachlabs.ai/api/v1/YOUR_FLOW_ID/trigger",
    headers={
        "X-API-KEY": os.getenv("EACH_API_KEY"), 
    },
    json={
        "parameters": {
            "text": "Hello World"
        }
    },
)

trigger_id = response.json().get("triggerID")

```

```go golang
import (
    each "github.com/eachlabs/eachgo"
)

func main() {
    client, err := each.NewClient(each.WithCredential("YOUR_API_KEY"))
    if err != nil {
        panic(err)
    }

    triggerID, err := client.TriggerFlow("flow-id", map[string]interface{}{
        "text": "Hello World",
    })
    if err != nil {
        panic(err)
    }

    fmt.Println(triggerID)

}
```

### Get Executions of a Flow

<CodeGroup>

```js javascript
// Please make sure that you run 'npm install @eachlabs/aiflow'
import Each from '@eachlabs/aiflow';

const each = new Each({
    auth: process.env.EACH_API_KEY ||
    'YOUR_API_KEY'
});

const executions = await each.flow.executions("flow-id");

console.log(executions)
```

```python python
# We will launch our python client SDK soon.
import requests

# Make sure that you set EACH_API_KEY environment variable

response = requests.get(
    "https://flows.eachlabs.ai/api/v1/YOUR_FLOW_ID/executions",
    headers={
        "X-API-KEY": os.getenv("EACH_API_KEY"), 
    },
)

executions = response.json()

```

```go golang
import (
    each "github.com/eachlabs/eachgo"

func main() {
    client, err := each.NewClient(each.WithCredential("YOUR_API_KEY"))
    if err != nil {
        panic(err)
    }

    executions, err := client.GetExecutions("flow-id")
    if err != nil {
        panic(err)
    }

    fmt.Println(executions)
}
```
